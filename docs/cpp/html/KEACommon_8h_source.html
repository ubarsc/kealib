256 inline std::string getDataTypeAsStr(KEADataType dataType)
257 {
258 std::string strDT = "Unknown";
259
260 if(dataType == kea_8int)
261 {
262 strDT = "Integer 8 bit";
263 }
264 else if(dataType == kea_16int)
265 {
266 strDT = "Integer 16 bit";
267 }
268 else if(dataType == kea_32int)
269 {
270 strDT = "Integer 32 bit";
271 }
272 else if(dataType == kea_64int)
273 {
274 strDT = "Integer 64 bit";
275 }
276 else if(dataType == kea_8uint)
277 {
278 strDT = "Unsigned Integer 8 bit";
279 }
280 else if(dataType == kea_16uint)
281 {
282 strDT = "Unsigned Integer 16 bit";
283 }
284 else if(dataType == kea_32uint)
285 {
286 strDT = "Unsigned Integer 32 bit";
287 }
288 else if(dataType == kea_64uint)
289 {
290 strDT = "Unsigned Integer 64 bit";
291 }
292 else if(dataType == kea_32float)
293 {
294 strDT = "Float 32 bit";
295 }
296 else if(dataType == kea_64float)
297 {
298 strDT = "Float 64 bit";
299 }
300 else
301 {
302 strDT = "Unknown";
303 }
304
305 return strDT;
306 }
